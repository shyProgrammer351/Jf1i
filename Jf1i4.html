알고리즘

프로그램시작 
   친구와! 대화하기 
      이름 물어보고 답하기 
      직업 물어보고 답하기 
      취미 물어보고 답하기 
      좋아하는 색 물어보고 답하기 
      친하게 지내자고 인사하기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter01/Chat.java 

	import java.lang.System;										 

	import java.util.Scanner;										 

	 

	public class Chat  

	{ 

		// 프로그램시작 

1		public static void main( String[] args )  						 

		{ 

			// 변수 선언 및 초기화 

			Scanner scan = new Scanner( System.in );				 

			String name = null;										 

			String job = null;										 

			String hobby = null;									 

2			String color = null;									 

	 

			// 친구와! 대화하기 

			// 이름 물어보고 답하기 

3			System.out.print( "안녕? 넌 이름이 뭐야: " );  

4			name = scan.nextLine();									 

5			System.out.println( "만나서 반가워, " + name + ". " );	 

	 

			// 직업 물어보고 답하기 

6			System.out.print( "넌 주로 뭐하고 살아: " );	 

7			job = scan.nextLine();									 

8			System.out.println( job + "? 나도 마찬가지야 ㅎㅎ " );	 

	 

			// 취미 물어보고 답하기 

9			System.out.print( "그럼 심심할 때 뭐하니: " ); 

10			hobby = scan.nextLine();								 

11			System.out.println( hobby + "? 나랑 똑같네. " ); 

	 

			// 좋아하는 색 물어보고 답하기 

12			System.out.print( "혹시 좋아하는 색은 뭐야 : " );	 

13			color = scan.nextLine();								 

14			System.out.println( color + "이라구? 와! 이것도 나랑 똑같네. 신기하다." ); 

	 

			// 친하게 지내자고 인사하기 

15			System.out.println( "앞으로 우리 친하게 지내자." ); 

	 

			// 프로그램 종료 

16			return; 

		} 

	}
