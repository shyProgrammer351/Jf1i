문제

음식을 주문 받는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
주문 도와드리겠습니다, 손님.
메인요리는 고르셨습니까 : 케이준치킨샐러드
소스는 어떻게 할까요 : 허니머스터드드레싱
음료는 무엇으로 준비할까요 : 콜라
메인요리는 케이준치킨샐러드, 소스는 허니머스터드드레싱, 음료는 콜라를 선택하셨습니다. 
준비하겠습니다, 잠시만 기다려 주십시오.

프로그램 코드

	import java.lang.System;							
	import java.util.Scanner;							
	
	public class FoodOrder 
	{
1		public static void main( String[] args )  						
		{
			Scanner scan = new Scanner( System.in );			
			String mainDish = null;						
			String sauce = null;						
2			String drink = null;						
	
3			System.out.println( "주문 도와드리겠습니다, 손님." );			
	
4			System.out.print( "메인요리는 고르셨습니까 : " );				
5			mainDish = scan.nextLine();								
	
6			System.out.print( "소스는 어떻게 할까요 : " );				
7			sauce = scan.nextLine();								
	
8			System.out.print( "음료는 무엇으로 준비할까요 : " );			
9			drink = scan.nextLine();								
	
10			System.out.println( "메인요리는 " + mainDish + ", 소스는 " + sauce + ", 음료는 " + drink + "를 선택하셨습니다. " );
11			System.out.println( "준비하겠습니다, 잠시만 기다려 주십시오." );	
	
12			return;
		}
	}
